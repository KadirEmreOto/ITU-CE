void setup() {
  //Initialize serial and wait for port to open:
  Serial.begin(9600);
  while (!Serial) {
    ; // wait for serial port to connect. Needed for native USB port only
  }


  pinMode(13,OUTPUT);
  digitalWrite(13,LOW);
  // prints title with ending line break
  Serial.println("Merhaba ! ");
}

String gelen;
void loop() {
  // if there's any serial available, read it:

    int gelen_baytlar=0;
    gelen_baytlar = Serial.available();

    gelen = Serial.readString();
    Serial.write(&gelen[0],1);
    Serial.write(&gelen[1],1);
    if ('a' == gelen[0])
      { 
         digitalWrite(13,HIGH);
         Serial.println("A:");
         
      }  
    
    
    if ('k' == gelen[1])
      { 
         digitalWrite(13,LOW);
         Serial.println("K:");
         
      }  
}
